-- Amigo Hospital Robot Delivery System - Supabase (PostgreSQL) Schema

-- 1. Wards / Facilities
CREATE TABLE wards (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

-- 2. User Profile (Staff / Admins)
-- Note: Authentication is skipped for now. We will map this to auth.users later.
CREATE TABLE user_profiles (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    display_name VARCHAR(100) NOT NULL,
    email VARCHAR(150) NOT NULL UNIQUE,
    role VARCHAR(50) NOT NULL DEFAULT 'Nurse', -- Admin, Nurse, etc.
    department VARCHAR(100),
    employee_id VARCHAR(50),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

-- 3. Robots (Amigo Fleet)
CREATE TABLE robots (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    ward_id BIGINT REFERENCES wards(id) ON DELETE SET NULL,
    current_location VARCHAR(50), -- e.g., 'Room 108', 'Base dock'
    status VARCHAR(50) NOT NULL DEFAULT 'Active', -- Active, En route, Charging, Maintenance
    battery_level INT NOT NULL DEFAULT 100,
    firmware_version VARCHAR(20),
    nfc_module_status VARCHAR(50) DEFAULT 'Online', -- Online, Fault
    line_follower_sensitivity INT DEFAULT 70,
    nfc_read_range INT DEFAULT 40,
    max_speed DECIMAL(3,2) DEFAULT 0.6,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

-- 4. Patients
CREATE TABLE patients (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    age INT,
    ward_id BIGINT REFERENCES wards(id) ON DELETE SET NULL,
    room_number VARCHAR(20),
    condition VARCHAR(255),
    nfc_tag VARCHAR(50) UNIQUE,
    assigned_robot_id BIGINT REFERENCES robots(id) ON DELETE SET NULL,
    status VARCHAR(20) DEFAULT 'Stable', -- Stable, Monitor, Critical
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

-- 5. Medicines (Inventory)
CREATE TABLE medicines (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    category VARCHAR(50), -- Analgesic, Antibiotic, Chronic care, etc.
    stock_quantity INT NOT NULL DEFAULT 0,
    unit VARCHAR(20) DEFAULT 'tabs', -- tabs, caps, pens
    reorder_level INT NOT NULL DEFAULT 50,
    expiry_date DATE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

-- 6. Deliveries (Dispense Schedule)
CREATE TABLE deliveries (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    patient_id BIGINT REFERENCES patients(id) ON DELETE CASCADE,
    medicine_id BIGINT REFERENCES medicines(id) ON DELETE CASCADE,
    assigned_robot_id BIGINT REFERENCES robots(id) ON DELETE SET NULL,
    scheduled_time TIMESTAMP WITH TIME ZONE NOT NULL,
    status VARCHAR(50) NOT NULL DEFAULT 'Scheduled', -- Scheduled, In Progress, Completed, Missed, Failed
    scanned_at TIMESTAMP WITH TIME ZONE, -- Time of NFC tap delivery confirmation
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

-- 7. Alerts & Notifications
CREATE TABLE alerts (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    type VARCHAR(50) NOT NULL, -- CRITICAL, WARNING, INFO
    message TEXT NOT NULL,
    is_resolved BOOLEAN DEFAULT false,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

-- 8. System Settings
CREATE TABLE system_settings (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    facility_name VARCHAR(200) NOT NULL,
    time_zone VARCHAR(100) NOT NULL,
    default_ward_id BIGINT REFERENCES wards(id) ON DELETE SET NULL,
    dark_theme BOOLEAN DEFAULT false,
    auto_refresh BOOLEAN DEFAULT true,
    auto_recall_low_battery BOOLEAN DEFAULT true,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);
